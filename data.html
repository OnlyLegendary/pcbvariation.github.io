from machine import Pin, ADC
import time

gpio_a = Pin(16, Pin.OUT)
gpio_b = Pin(17, Pin.OUT)

adc_a = ADC(26)
adc_b = ADC(27)

THRESH = 20000

def measure(gpio, adc):
    gpio.value(0)
    time.sleep_us(200)
    start = time.ticks_us()
    gpio.value(1)
    while adc.read_u16() < THRESH:
        pass
    return time.ticks_diff(time.ticks_us(), start)
